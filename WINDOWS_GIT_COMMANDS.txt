# ========================================
# GIT COMMANDS FOR WINDOWS
# Copy and paste these ONE AT A TIME
# ========================================

# Step 1: Initialize Git
git init

# Step 2: Add all files
git add .

# Step 3: Commit (SINGLE LINE - works on Windows)
git commit -m "feat: Smart Irrigation AI Agent with 98.70% accuracy - 6 ML models with gradient descent"

# Step 4: Add remote (REPLACE YOUR_USERNAME with your actual GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/smart-irrigation-ai-agent.git

# Step 5: Rename branch to main
git branch -M main

# Step 6: Push to GitHub
git push -u origin main

# ========================================
# IMPORTANT NOTES:
# ========================================
# 1. Replace YOUR_USERNAME with your GitHub username
# 2. When prompted for password, use Personal Access Token (NOT your GitHub password)
# 3. Run each command ONE AT A TIME
# 4. Wait for each command to complete before running the next one
